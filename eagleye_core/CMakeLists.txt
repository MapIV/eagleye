cmake_minimum_required(VERSION 2.8.3)
project(eagleye_core)

find_package(catkin REQUIRED COMPONENTS
  roscpp
  std_msgs
  geometry_msgs
  sensor_msgs
  rtklib_msgs
  eagleye_msgs
  coordinate
)

   CATKIN_DEPENDS
   roscpp
   std_msgs
   geometry_msgs
   sensor_msgs
   rtklib_msgs
   eagleye_msgs

)

  include

)
